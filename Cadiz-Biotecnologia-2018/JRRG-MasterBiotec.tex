\begin{frame}{Chemotaxis}
%---------------------------------------------------------------------
  \begin{itemize}\itemsep1em
  \item Movement of \alert{biological cells} in response
    to \alert{chemical signals}
    \begin{quotation}
      $\rightarrow$
      \href{https://youtu.be/ZUUfdP87Ssg?t=37}{Video: neutrophile chemotaxis}
    \end{quotation}
    \pause
  \item  \emph{General formulation} (\structure{Keller--Segel} 1970')~\cite{winkler_bellomo_toward_2015}
    \par\bigskip
    \begin{quotation}
    Find \alert{$u$}, \alert{$v$} (\underline{\textit{density of cells}} and
    \underline{\textit{chemical-signal}}) such that:
    \vspace{-0.3em}
  \begin{equation*}\def\arraystretch{1.2}
    \label{K-S}
    \left\{
      \begin{array}{ll@{}}
        u_t= \div\Big( D_u(u,v)\*\grad u - \chi(u,v) \* u \* \nabla v\Big) + H(u,v), \\
        v_t= D_v(u,v)\*\Delta v + K(u,v)
      \end{array}\right.
  \end{equation*}
\end{quotation}\vspace{-1.0em}
\item \textbf{Many variants}\footnote{See e.g.~\cite{horstmann_2003}
    for a review}  depending on (likely nonlinear) terms...
  \par\smallskip
  \begin{itemize}\itemsep0.66em
  \item \structure{Diffusion} of cells and \alert{chemoattractant}:  $D_u(u,v)$, $D_v(u,v)$
  \item Source terms: $H(u,v)$, $K(u,v)$
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Mathematical Properties}
%---------------------------------------------------------------------
  A general result (local well-posedness):
  \medskip
  \begin{itemize}\itemsep1.33em
  \item If \structure{$u_0\in C^0(\overline\Omega)$},\enspace
    \structure{$v_0\in W^{1,q}(\Omega)$} \small non-negative \gris{($q>n=\mbox{space
      dim.}$)}
  \item Then \enspace \llaveizq{$\structure{\exists \Tmax} \in(0,+\infty]$ and\\[0.9ex]
    $\alert{\exists!u, v}\in C^0\big(\overline\Omega\times
      (0,\Tmax)\big)$} \quad such that
    \medskip
    \begin{itemize}\itemsep0.95em
    \item $(u,v)$ \alert{solves classical} \alert{Keller-Segel} equations in $\Omega\times (0,\Tmax)$
    \item \alert{$u,v\ge 0$} (positivity of solution)
    \item If $\structure{\Tmax<+\infty}$ $\Rightarrow$
      $\alert{\norm[L^\infty(\Omega)]{u}} +
      \alert{\norm[W^{1,q}(\Omega)]{v}}\to\infty$ as $t\nearrow\Tmax$
  \end{itemize}
  \item Other mathematical properties: mass conservation, energy law...
  \end{itemize}
\end{frame}

\begin{frame}{The Classical Keller-Segel System}
%---------------------------------------------------------------------
  We focus on a \structure{minimal} system in a space domain 2D or 3D:

  \par\bigskip\quad
  Find \structure{$u$}, (\textit{density of cells}) and \structure{$v$}
  (\textit{chemical-signal}) such that:
  \begin{quotation}
    \begin{equation*}\def\arraystretch{1.2}
      \label{K-S}
      \left\{
        \begin{array}{ll@{}}
          u_t= \kA\Delta u - \alert{\kB\nabla\cdot( u\nabla v)}, &  \quad {\color{gray}x\in\Omega,\, t>0,}\\
          v_t= \kC\Delta v - \kD v + \kE u, &  \quad {\color{gray}x\in\Omega,\, t>0.}
        \end{array}\right.
    \end{equation*}
  \end{quotation}
  \begin{itemize}\itemsep0.66em
  \item Parameters $\alpha_i>0$
  \item \alert{Nonlinear chemotaxis term}! {\color{gray}Hyperbolic effects expected if $\alpha_1\!\gg$}
  \item Initial conditions (important \structure{rule in long-time} behaviour)
  \end{itemize}

\end{frame}

\begin{frame}{Time Discretization}
%---------------------------------------------------------------------
  We fix a discrete time partition
  \medskip
  \begin{itemize}
  \item $0=t_0<t_1<\cdots<t_N=T$, \quad {\color{gray}$k=\tmm-\tm$}
  \item
    $\um\approx u(\tm)$, \quad{\color{gray}$\forall m=0,...,N$}
  \end{itemize}
  \bigskip
  \begin{block}{\textbf{\color{white}{Euler Time Scheme Family}}}
    \vspace*{-0.5em}
    \begin{large}
  \begin{align*}
    (1/k)\umm &- \Delta\umm + \div(u^{\alert{m+s_1}}\grad v^{\alert{m+s_2}} ) = (1/k)\um
    \\[0.7em]
    (1/k)\vmm &- \Delta\vmm + v^{\alert{m+s_3}} - u^{\alert{m+s_4}} = (1/k)\vm
  \end{align*}
\end{large}
  % where $r_1$, $r_2$, $r_3$, $r_4$ $\in\{0,1\}$,
  where ${S}=(s_1,s_2,s_3,s_4)\in\{0,1\}^4$.
\end{block}
\end{frame}
